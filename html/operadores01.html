Ejercicio: Operadores aritmáticos

let a = 5;
let b = 7;

document.write(a + b, "suma"); //suma
document.write(a - b, "resta"); //resta
document.write(a * b, "producto"); //multiplicacion
document.write(a / b, "división"); //división
document.write(a % b, "módulo"); //modulo
document.write(a ** b, "potencia"); //potencia

//incrementar
document.write("incrementar")
document.write(a, "a");
document.write(++a, "++a");
document.write(a, "a");
document.write(a++,"a++");
document.write(a, "a");

//decrementar
document.write("decrementar")
document.write(b, "b");
document.write(--b, "--b");
document.write(b, "b");
document.write(b--,"b--");
document.write(b, "b");
