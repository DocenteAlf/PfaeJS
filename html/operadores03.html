Ejercicio: Operadores de comparación

let a=10; document.write (a," a");
//relacionales
document.write (a>5," a>5");
document.write (a>=5," a>=5");
document.write (a>=10," a>=10");
document.write (a<5," a<5");
document.write (a<=5," a<=5");

//de igualdad
document.write (a==10," a==10");
document.write (a!=10," a!=10");
document.write (a=='10'," a=='10'"); //pregunta si los valores son inguales
document.write (a!='10'," a!='10'");
